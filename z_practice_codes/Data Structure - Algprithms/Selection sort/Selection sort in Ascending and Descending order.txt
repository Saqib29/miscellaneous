#include <iostream>
using namespace std;
class Selection_sort
{
    int A[10];
public:
    void set_data();
    void Ascending();
    void Descending();
    void output();
};

void Selection_sort::set_data()
{
    cout << "Put the values: " << endl;
    for(int i = 0; i<10; i++)
    {
        cin >> A[i];
    }
}

void Selection_sort::Ascending()
{
    int i, j, temp, min;
    for(i=0; i<10; i++)
    {
        min = i;
        for(j=i+1; j<10; j++)
        {
            if(A[min] > A[j])
            {
                min = j;
            }
        }
        if(min != i)
        {
            swap(A[min], A[i]);
        }
    }
}

void Selection_sort::Descending()
{
    int i, j, temp, max;
    for(i=0; i<10; i++)
    {
        max = i;
        for(j=i+1; j<10; j++)
        {
            if(A[max] < A[j])
            {
                max = j;
            }
        }
        if(max != i)
        {
            swap(A[max], A[i]);
        }
    }
}

void Selection_sort::output()
{

    cout << endl;
    int i;
    for(i=0; i<10; i++)
    {
        cout << A[i] << endl;
    }
}

int main()
{
    Selection_sort ob1, ob2;

    cout << "\t\tInsertion Sort!!" << endl << endl;

    cout << "For Ascending order...." << endl;
    ob1.set_data();
    ob1.Ascending();
    ob1.output();
    cout << "Complete Ascending order!" << endl;

    cout << "For Descending order..." << endl;
    ob2.set_data();
    ob2.Descending();
    ob2.output();
    cout << "Complete Descending order!" << endl;
    return 0;
}
